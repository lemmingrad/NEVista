//-- Hungarian notation
// g	global variable
// m	member of a class
// sm	static member of class
// p	pointer
// n	integer number, signed or unsigned
// f	floating point number, signed, unsigned, float or double
// b	boolean
// a	array
// bit	bit flags. Use bit instead of n for bit variables
// str	string. 
//			Use s8 *strSomething instead of s8 *pSomething
//			use s8 strSomething instead of s8 aSomething[]
//			Use s8 **pstrSomething instead of s8 *ppSomething or s8 *pSomething[]

// x, y, z always considered f32
// r, g, b, a always considered u8

//-- Types definition names. Not variable names.
// sXXX		struct
// uXXX		union
// eXXX		enum
// clXXX	class

//-- Singleton Managers - Do NOT use constructor and destructor. 
// Init()
// Shutdown()

//-- Other Objects
// Create()
// Destroy()

//-- Where memory is involved
// Allocate()
// Delete()

//-- Where lists are involved
// Ins()
// Remove()
